CREATE TABLE prodebians (
 id_prodebian   serial PRIMARY KEY,
 title          varchar(64),
 debversion     real,
 version        integer,
 property       bit varying(4),
 id_owner       integer,
 description    text,
 actionlist     integer[]
);

CREATE TABLE owners (
 id_owner       serial PRIMARY KEY,
 username       varchar(64) UNIQUE,
 email          varchar(64),
 password       varchar(64),
 description text
);

CREATE TABLE actions (
 id_action      serial PRIMARY KEY,
 title          varchar(64),
 actiontype  integer,
 description text,
 actionvalues   text[]
);

CREATE TABLE packages (
 id_pack      serial PRIMARY KEY,
 description text,
 pack_name    varchar(64) UNIQUE,
 pack_version varchar(16),
 pack_release varchar(16)
);
